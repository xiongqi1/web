475 /* Return value in dB*1000 (mdB) */
476 static int32 dBLookup(uInt32 number)
477 {
478  int i;
479  uInt32 err;
480 
481  if(number >= dBTable10_n60[0])
482  {
483  return 10000; /* out of range - clamp at 10dB */
484  }
485 
486  for(i=0; i<MAX_DB_TABLE; i++) /* 139 */
487  {
488  if((number < dBTable10_n60[i])&&(number >= dBTable10_n60[i+1]))
489  {
490  /* See which level it is closest to */
491  err = dBTable10_n60[i] - number;
492  if(err < (number - dBTable10_n60[i+1]))
493  {
494  return (10000 - i*DB_TABLE_STEP_SIZE);
495  }
496  else
497  {
498  return (10000 - (i+1)*DB_TABLE_STEP_SIZE);
499  }
500  }
501  }
502  /* No solution found? Return -60dB */
503  return -60000;
504 }

567 static uInt32 Isqrt32(uInt32 num)
568 {
569  uInt32 one,res,op;
570 
571  op=num;
572  res=0;
573  one = 1L << 30;
574  while(one > op)
575  {
576  one >>=2;
577  }
578 
579  while(one !=0)
580  {
581  if(op >= res + one)
582  {
583  op = op - (res + one);
584  res = res + 2*one;
585  }
586  res >>= 1;
587  one >>= 2;
588  }
589 
590  return (res);
591 }

662  proslicTestInObjType *pTstin)
663 {
664  uInt8 regData;
665 
666  /* Valid device check */
667  if(TSTIN_INVALID_PART_NUM)
668  {
669  return RC_UNSUPPORTED_FEATURE;
670  }
671 
672  /* Check if enabled */
673  if(!pTstin->pcmLpbkTest.testEnable)
674  {
675  return RC_TEST_DISABLED;
676  }
677  /* Return if already enabled */
678  if(pTstin->pcmLpbkTest.pcmLpbkEnabled)
679  {
680  return RC_NONE;
681  }
682 
683  /* Store PCM Settings */
684  pTstin->pcmLpbkTest.pcmModeSave = ProSLIC_ReadReg(pProslic,11); /* PCMMODE */
685 
686 
687  /* Disable PCM bus before changing format */
688  regData = pTstin->pcmLpbkTest.pcmModeSave & ~0x10; /* PCM_EN = 0 */
689  ProSLIC_WriteReg(pProslic,11,regData);
690 
691  /* Configure for either 8 or 16bit linear */
692  if(pTstin->pcmLpbkTest.pcm8BitLinear == PCM_8BIT)
693  {
694  regData |= 0x02; /* PCM_FMT[1] = 1 */
695  regData &= ~0x01; /* PCM_FMT[0] = 0 */
696  }
697  else /* PCM_16BIT */
698  {
699  regData |= 0x03; /* PCM_FMT[1:0] = 11 */
700  }
701 
702  ProSLIC_WriteReg(pProslic,11,regData);
703 
704  /* Enable PCM Loopback */
705  ProSLIC_WriteReg(pProslic,43,0x01); /* LOOPBACK */
706 
707  /* Re-enable PCM Bus */
708  ProSLIC_WriteReg(pProslic,11,regData|0x10); /* PCMMODE */
709 
710  pTstin->pcmLpbkTest.pcmLpbkEnabled = 1;
711  DEBUG_PRINT(pProslic, "ProSLIC : TestIn : pcmLpbk : ENABLED\n");
712  return RC_NONE;
713 }
