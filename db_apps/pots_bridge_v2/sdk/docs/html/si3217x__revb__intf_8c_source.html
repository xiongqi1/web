98 int Si3217x_RevB_GenParamUpdate(proslicChanType_ptr pProslic,initSeqType seq)
99 {
100  uInt8 data;
101  ramData ram_data;
102  uInt8 oi_sense_scale;
103 
104  switch(seq)
105  {
106  case INIT_SEQ_PRE_CAL:
107  /*
108  ** Force pwrsave off and disable AUTO-tracking - set to user configured state after cal
109  */
110  WriteReg(pProHW, pProslic->channel,SI3217X_COM_REG_ENHANCE,0);
111  WriteReg(pProHW, pProslic->channel,SI3217X_COM_REG_AUTO,0x2F);
112 
113  /*
114  ** General Parameter Updates
115  */
116  WriteRAM(pProHW, pProslic->channel,SI3217X_REVB_RAM_DCDC_FSW_VTHLO,
117  Si3217x_General_Configuration.dcdc_fsw_vthlo);
118  WriteRAM(pProHW, pProslic->channel,SI3217X_REVB_RAM_DCDC_FSW_VHYST,
119  Si3217x_General_Configuration.dcdc_fsw_vhyst);
120  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_P_TH_HVIC,
121  Si3217x_General_Configuration.p_th_hvic);
122  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_COEF_P_HVIC,
123  Si3217x_General_Configuration.coef_p_hvic);
124  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_BAT_HYST,
125  Si3217x_General_Configuration.bat_hyst);
126  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_VBATH_EXPECT,
127  Si3217x_General_Configuration.vbath_expect);
128  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_VBATR_EXPECT,
129  Si3217x_General_Configuration.vbatr_expect);
130  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_PWRSAVE_TIMER,
131  Si3217x_General_Configuration.pwrsave_timer);
132  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_OFFHOOK_THRESH,

137  Si3217x_General_Configuration.vbat_track_min_rng);
138  WriteRAM(pProHW, pProslic->channel,SI3217X_REVB_RAM_DCDC_FSW_NORM,
139  Si3217x_General_Configuration.dcdc_fsw_norm);
140  WriteRAM(pProHW, pProslic->channel,SI3217X_REVB_RAM_DCDC_FSW_NORM_LO,
141  Si3217x_General_Configuration.dcdc_fsw_norm_lo);
142  WriteRAM(pProHW, pProslic->channel,SI3217X_REVB_RAM_DCDC_FSW_RINGING,
143  Si3217x_General_Configuration.dcdc_fsw_ring);

152  Si3217x_General_Configuration.pd_uvlo);

168  Si3217x_General_Configuration.dcdc_ana_tonmax);
169 
170 
171  /*
172  ** Hardcoded RAM
173  */
174 
175  /* OITHRESH */
176  if( Si3217x_General_Configuration.bom_option == BO_DCDC_LCQC_3W)
177  {
178  oi_sense_scale =
179  2; /* OITHRESH should be twice as large as user input when 0.2 ohm Rsense is used -LG, 07/18/14 */
180  }
181  else
182  {
183  oi_sense_scale = 1;
184  }
185 
186  ram_data = (oi_sense_scale*GCONF.i_oithresh > OITHRESH_OFFS)?
187  (oi_sense_scale*GCONF.i_oithresh - OITHRESH_OFFS)/OITHRESH_SCALE:0L;
188  ram_data *= BIT20LSB;
189  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_OITHRESH,ram_data);
190 
191 
192  /* OVTHRESH */
193  ram_data = (GCONF.v_ovthresh > OVTHRESH_OFFS)?(GCONF.v_ovthresh -
194  OVTHRESH_OFFS)/OVTHRESH_SCALE:0L;
195  ram_data *= BIT20LSB;
196  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_OVTHRESH,ram_data);

200  UVTHRESH_OFFS)/UVTHRESH_SCALE:0L;
201  ram_data *= BIT20LSB;
202  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_UVTHRESH,ram_data);
203 
204  /* UVHYST */
205  ram_data = (GCONF.v_uvhyst > UVHYST_OFFS)?(GCONF.v_uvhyst -
206  UVHYST_OFFS)/UVHYST_SCALE:0L;
207  ram_data *= BIT20LSB;
208  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_UVHYST,ram_data);
209 
210  /* PM BOM dependent audio gain scaling */
211  if(Si3217x_General_Configuration.pm_bom == BO_PM_BOM)
212  {
213  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_SCALE_KAUDIO,BOM_KAUDIO_PM);
214  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_AC_ADC_GAIN,
215  BOM_AC_ADC_GAIN_PM);
216  }
217  else
218  {
219  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_SCALE_KAUDIO,
220  BOM_KAUDIO_NO_PM);
221  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_AC_ADC_GAIN,
222  BOM_AC_ADC_GAIN_NO_PM);
223  }
224 
225  /* Default GPIO config - coarse sensors */
226  data = ReadReg(pProHW, pProslic->channel,SI3217X_COM_REG_GPIO_CFG1);
227  data &= 0xF9; /* Clear DIR for GPIO 1&2 */
228  data |= 0x60; /* Set ANA mode for GPIO 1&2 */
229  WriteReg(pProHW,pProslic->channel,SI3217X_COM_REG_GPIO_CFG1,
230  data); /* coarse sensors analog mode */
231 
232 
233  /* Misc modifications to default settings */
234  WriteReg(pProHW,pProslic->channel,SI3217X_COM_REG_PDN,
235  0x80); /* madc powered in open state */
236  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_TXACHPF_A1_1,
237  0x71EB851L); /* Fix HPF corner */
238  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_ROW0_C2,
239  0x723F235L); /* improved DTMF det */
240  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_ROW1_C2,
241  0x57A9804L); /* improved DTMF det */
242  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_PD_REF_OSC,
243  0x200000L); /* PLL freerun workaround */
244  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_ILOOPLPF,
245  0x4EDDB9L); /* 20pps pulse dialing enhancement */
246  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_ILONGLPF,
247  0x806D6L); /* 20pps pulse dialing enhancement */
248  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_VDIFFLPF,
249  0x10038DL); /* 20pps pulse dialing enhancement */
250  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_VCMLPF,
251  0x10059FL); /* 20pps pulse dialing */
252  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_CM_SPEEDUP_TIMER,
253  0xF0000L); /* 20pps pulse dialing */
254  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_VCM_HYST,0x206280L);
255  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_VCM_TH,

257  WriteRAM(pProHW,pProslic->channel,SI3217X_REVB_RAM_LKG_STBY_OFFSET,

278  WriteRAM(pProHW,pProslic->channel,PSR_DCDC_OVTHRESH_NEW,0x700000L);

282  WriteRAM(pProHW,pProslic->channel,PSR_DCDC_OVTHRESH_NEW,0x300000L);

302  break;
303 
304  }

337 #else
338  *patch = &(SI3217X_PATCH_B_BKBT);

343 #ifdef SIVOICE_NEON_MWI_SUPPORT /* PATCH DOES NOT SUPPORT MWI */

345  pProslic->error = RC_INVALID_PATCH;
346  return RC_INVALID_PATCH;
347 #else
348  *patch = &(SI3217X_PATCH_B_PBB);

372 /*
373 ** Function: Si3217x_RevB_ConverterSetup
374 **
375 ** Description:
376 ** Program revision specific settings before powering converter
377 **
378 ** Specifically, from general parameters and knowledge that this
379 ** is Rev B, setup dcff drive and gate drive polarity.
380 **
381 ** Returns:
382 ** int (error)
383 **
384 */
385 int Si3217x_RevB_ConverterSetup(proslicChanType_ptr pProslic)
386 {
387  ramData inv_on;
388  ramData inv_off;
389 
390  /* Option to add a per-channel inversion for maximum flexibility */
391  if(pProslic->dcdc_polarity_invert)
392  {
393  inv_on = 0x0L;
394  inv_off = 0x100000L;
395  }
396  else
397  {
398  inv_on = 0x100000L;
399  inv_off = 0x0L;
400  }
401 
402  switch(Si3217x_General_Configuration.bom_option)
403  {
404  case BO_DCDC_FLYBACK:
405  case BO_DCDC_LCQC_3W:
406  /*
407  ** All revB flyback designs have gate driver,
408  ** inverted polarity, and no dcff drive
409  */
410  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_DCFF_ENABLE,0x0L);
411  WriteRAM(pProHW, pProslic->channel,SI3217X_REVB_RAM_DCDC_SWFET,0x300000L);
412  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_SWDRV_POL,inv_on);
413  break;
414 
415  case BO_DCDC_BUCK_BOOST:
416  /*
417  ** All revB bjt buck boost designs have dcff drive,
418  ** no gate driver, and non-inverted polarity
419  */
420  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_DCFF_ENABLE,
421  0x10000000L);
422  WriteRAM(pProHW, pProslic->channel,SI3217X_REVB_RAM_DCDC_SWFET,0x300000L);
423  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_SWDRV_POL,inv_off);
424  break;
425 
426  case BO_DCDC_PMOS_BUCK_BOOST:
427  /*
428  ** RevB pmos buck boost designs have gate driver,
429  ** non-inverted polarity, and dcff drive on low voltage option only
430  */
431 
432  if(Si3217x_General_Configuration.vdc_range == VDC_3P0_6P0)
433  {
434  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_DCFF_ENABLE,
435  0x10000000L);
436  }
437  else
438  {
439  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_DCFF_ENABLE,0x0L);
440  }
441  WriteRAM(pProHW, pProslic->channel,SI3217X_REVB_RAM_DCDC_SWFET,0x300000L);
442  WriteRAM(pProHW, pProslic->channel,SI3217X_COM_RAM_DCDC_SWDRV_POL,inv_off);
443  break;
444 
445  default:
446  return RC_DCDC_SETUP_ERR;
447  }
448 
449  return RC_NONE;
450 }
