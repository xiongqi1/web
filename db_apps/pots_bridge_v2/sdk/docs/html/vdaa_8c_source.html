41 ** Function: isVerifiedDAA
42 **
43 ** Description:
44 ** Verifies addressed channel is DAA
45 **
46 ** Input Parameters:
47 ** pVdaa: pointer to SiVoiceChanType or vdaaChanType
48 **
49 ** Return:
50 ** Verified DAA
51 ** Not DAA RC_CHANNEL_TYPE_ERR
52 **
53 */
54 
55 static int isVerifiedDAA(vdaaChanType *pVdaa)
56 {
57  uInt8 data = ReadReg(pVdaaHW,pVdaa->channel,LSIDE_REV);
58  if ( (data & 0x40) == 0 ) /*This bit is always 1 for DAA*/
59  {
60  LOGPRINT("%sDAA device not detected\n",LOGPRINT_PREFIX);
61  return RC_CHANNEL_TYPE_ERR;
62  }
63  else
64  {
65  /* For Si3050, the value will be zero initially (calloc), for Si32178/9, this will
66  * be non-zero (Rev B).
67  */
68  if(pVdaa->deviceId->chipRev == 0)
69  {
70  /* Read Device ID and store it */
71  /* NOTE: in earlier releases we also read the line side info here. This is now done
72  Vdaa_duringPowerUpLineside since the information we read was always 0.
73  */
74  data = pVdaa->ReadRegX(pVdaa->pVdaaHWX,pVdaa->channel,SYS_LINE_DEV_REV);
75  pVdaa->deviceId->chipRev= data&0xF;
76 #ifdef ENABLE_DEBUG
77  LOGPRINT("%sChipRev = 0x%x\n", LOGPRINT_PREFIX, pVdaa->deviceId->chipRev);
78 #endif
79  }
80  }
81 
82  return RC_NONE;
83 }

309 ** Function: Vdaa_PCMTimeSlotSetup
310 **
311 ** Description:
312 ** configure pcm timeslot
313 **
314 ** Returns:
315 **
316 */
317 int Vdaa_PCMTimeSlotSetup (vdaaChanType *pVdaa, uInt16 rxcount, uInt16 txcount)
318 {
319  uInt8 data = 0;
320  uInt8 pcmStatus;
321 
322  TRACEPRINT( pVdaa, "rxcount = %u txcount = %u\n", (unsigned int)rxcount,
323  (unsigned int)txcount);
324  if(pVdaa->channelType != DAA)
325  {
326  return RC_IGNORE;
327  }
328 
329  /* Disable PCM if enabled - restore after updating timeslots */
330  pcmStatus = ReadReg(pVdaaHW,pVdaa->channel,PCM_SPI_CTRL);
331  if (pcmStatus&0x20)
332  {
333  WriteReg(pVdaaHW,pVdaa->channel,PCM_SPI_CTRL,pcmStatus&~(0x20));
334  }
335 
336  /*Storing 10 bit value of Transmit PCM sample in REG 34 and REG 35[0:1]*/
337  data = (uInt8)(txcount & 0xff);
338  WriteReg(pVdaaHW,pVdaa->channel,PCMTX_CNT_LO,data);
339  data = (uInt8)(txcount >> 8) ;
340  WriteReg(pVdaaHW,pVdaa->channel,PCMTX_CNT_HI,data);
341 
342  /*Storing 10 bit value of Receive PCM sample in REG 34 and REG 35[0:1]*/
343  data = (uInt8)(rxcount & 0xff);
344  WriteReg(pVdaaHW,pVdaa->channel,PCMRX_CNT_LO,data);
345  data = (uInt8)(rxcount >> 8);
346  WriteReg(pVdaaHW,pVdaa->channel,PCMRX_CNT_HI,data);
347 
348  /* Enable back the PCM after storing the values*/
349  WriteReg(pVdaaHW,pVdaa->channel,PCM_SPI_CTRL,pcmStatus);
350 
351  return RC_NONE;
352 }

492  break;

497

499  regData = 0x88;
500  break;
501  }
502 
503  WriteReg(pVdaaHW,pVdaa->channel,TXRX_GN_CTRL,regData);

641  if(pVdaa[k]->channelType == PROSLIC)
642  {
643  continue; /* Skip if we know this is a ProSLIC, else check the device out */
644  }
645 
646  if (isVerifiedDAA(pVdaa[k]) == RC_CHANNEL_TYPE_ERR)
647  {
648  pVdaa[k]->channelEnable = FALSE;
649  pVdaa[k]->error = RC_CHANNEL_TYPE_ERR;
650  pVdaa[k]->channelType = UNKNOWN;
651  DEBUG_PRINT(pVdaa[k], "%sVDAA not supported on this device\n",LOGPRINT_PREFIX);
652  continue;
653  }
654  else
655  {
656  pVdaa[k]->channelType = DAA;
657  }
658 
659  if (pVdaa[k]->channelEnable)
660  {
661  /*Try to write innocuous register to test SPI is working*/
662  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,PCMRX_CNT_LO,0x5A);
663  data = pVdaa[k]->ReadRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,PCMRX_CNT_LO);
664  if (data != 0x5A)
665  {
666  pVdaa[k]->error = RC_SPI_FAIL;
667  pVdaa[k]->channelEnable = FALSE;
668  DEBUG_PRINT(pVdaa[k], "%sVDAA %d not communicating\n",LOGPRINT_PREFIX,
669  pVdaa[k]->channel);
670  }
671  else
672  {
673  num_devices++;
674  }
675  }
676  }
677 
678  if(num_devices == 0)
679  {
680  DEBUG_PRINT(*pVdaa, "%sNo DAA devices detected\n", LOGPRINT_PREFIX);
681  return RC_SPI_FAIL;
682  }
683 
684  for (k=0; k<size; k++)
685  {
686  if(pVdaa[k]->channelType != DAA)
687  {
688  continue; /* Skip PROSLIC or UNDEFINED ports */
689  }
690  if (pVdaa[k]->channelEnable)
691  {
692 
693  /* Apply General Configuration parameters */
694 
695  /* No need to read-modify-write here since registers are in their reset state */
696  data = (Vdaa_General_Configuration.pwmm << 4) |
697  (Vdaa_General_Configuration.pwmEnable << 3);
698  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,CTRL1, data);
699 
700  data = (Vdaa_General_Configuration.inte << 7) | (Vdaa_General_Configuration.intp
701  << 6) | 0x03;
702  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,CTRL2, data);
703 
704  data = (Vdaa_General_Configuration.hssm << 3);
705  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,SMPL_CTRL, data);
706 
707  data = (Vdaa_General_Configuration.iire << 4);
708  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,INTL_CTRL1, data);
709 
710  data = (Vdaa_General_Configuration.rcald << 4);
711  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,RES_CALIB, data);
712 
713  data = (Vdaa_General_Configuration.full2 << 4);
714  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,AC_TERM_CTRL, data);
715 
716  data = (Vdaa_General_Configuration.lvfd) | (Vdaa_General_Configuration.filt <<
717  1) |
718  (Vdaa_General_Configuration.foh << 5) | (Vdaa_General_Configuration.full << 7);
719  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,DAA_CTRL5, data);
720 
721  data = (Vdaa_General_Configuration.spim << 6);
722  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,PCM_SPI_CTRL, data);
723 
724  data = (Vdaa_General_Configuration.cvp) | (Vdaa_General_Configuration.cvs << 2);
725  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,LN_VI_THRESH_INTE_CTRL,
726  data);
727 
728  data = (Vdaa_General_Configuration.gce << 1) | (Vdaa_General_Configuration.rg1
729  << 2);
730  pVdaa[k]->WriteRegX(pVdaa[k]->pVdaaHWX,pVdaa[k]->channel,SPRK_QNCH_CTRL, data);
731 
732  /* Enable Lineside Device */
733  Vdaa_PowerupLineside(pVdaa[k]);
734  }
735  }
736  DEBUG_PRINT(*pVdaa,"%sDAA initialization completed\n",LOGPRINT_PREFIX);
737  return RC_NONE;
738 }

909 
910 int Vdaa_SetHookStatus (vdaaChanType *pVdaa,uInt8 newHookStatus)
911 {

948  WriteReg(pVdaaHW,pVdaa->channel,DAA_CTRL1,data);

952  break;

954  case VDAA_ONHOOK_MONITOR:
955  /*Setting REG6[4]=0,REG5[0]=0,REG5[3]=1*/

959  WriteReg(pVdaaHW,pVdaa->channel,DAA_CTRL1,data);

961  data &= 0xEF;
962  WriteReg(pVdaaHW,pVdaa->channel,DAA_CTRL2,data);
963  break;
964 
965  default:

967  }
