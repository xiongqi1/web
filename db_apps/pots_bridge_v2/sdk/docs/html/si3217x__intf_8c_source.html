115 /*
116 ** Function: getChipType
117 **
118 ** Description:
119 ** Decode ID register to identify chip type
120 **
121 ** Input Parameters:
122 ** ID register value
123 **
124 ** Return:
125 ** partNumberType
126 */
127 static partNumberType getChipType(uInt8 data)
128 {
129  /* For the parts that have a HV variant, we map to the lower voltage version,
130  the actual differences are handled in the constants file
131  */
132  const uInt8 revBPartNums[8] =
133  {
134  SI32171, UNSUPPORTED_PART_NUM, UNSUPPORTED_PART_NUM, UNSUPPORTED_PART_NUM,
135  SI32176, SI32177, SI32178, UNSUPPORTED_PART_NUM
136  };
137 
138  const uInt8 revCPartNums[8] =
139  {
140  SI32171, UNSUPPORTED_PART_NUM, UNSUPPORTED_PART_NUM, SI32175,
141  SI32176, SI32172, SI32174, UNSUPPORTED_PART_NUM
142  };
143 
144  uInt8 partNum = (data & 0x38) >> 3;
145  if( (data & 0x7) ==1 ) /* Rev B */
146  {
147  return revBPartNums[ partNum ];
148  }
149 
150  if ( (data & 0x7) == 2 ) /* Rev C */
151  {
152  return revCPartNums[ partNum ];
153  }
154 
155  return UNSUPPORTED_PART_NUM;
156 }

158 /*
159 ** Function: Si3217x_GetChipInfo
160 **
161 ** Description:
162 ** Determine if DAA or ProSLIC present
163 **
164 ** Input Parameters:
165 ** pProslic: pointer to PROSLIC channel object
166 **
167 ** Return:
168 ** channelType
169 */
170 int Si3217x_GetChipInfo(proslicChanType *pProslic)
171 {
172  uInt8 data, id;
173  uInt8 daaEnableSave = 0;
174  int returnCode;
175 
176  /*
177  ** Set bit 0 of DAA_CNTL reg regardless of channel
178  ** type. This will allow us to see the DAA register
179  ** space if an Si32178 or Si3050 occupies that channel.
180  ** If it is a DAA channel, the register that is actually
181  ** written is FXO Control 3, since the address bit 6 is
182  ** ignored. This sets DDL (digital loopback). This should
183  ** be OK provided it is returned to it's entry state once
184  ** the channel type has been identified.
185  **
186  **
187  */
188 
189  /*
190  ** Note: DIAG3 replaced DAA_CNTL on Si3217x RevC. Setting DIAG3.0
191  ** sets PD_GR909, which powers down the HVIC. Must pre-check
192  ** the device revision before doing DAA check.
193  */
194  id = ReadReg(pProHW,pProslic->channel,SI3217X_COM_REG_ID);
195  if(id&0x02)
196  {
197  pProslic->channelType = PROSLIC;
198  pProslic->deviceId->chipRev = id & 0x7;
199  pProslic->deviceId->chipType = getChipType(id);
200 
201  if(pProslic->deviceId->chipType == UNSUPPORTED_PART_NUM)
202  {
203 #ifdef ENABLE_DEBUG
204  LOGPRINT("%sregister 0 read = 0x%02X\n", LOGPRINT_PREFIX, id);
205 #endif
206  return RC_SPI_FAIL;
207  }
208  else
209  {
210  return RC_NONE;
211  }
212  }
213 
214  /* If the DAA control register was enabled, then check if this is the
215  correct setting.
216  */
217  if(Si3217x_General_Configuration.daa_cntl != 0)
218  {
219  daaEnableSave = ReadReg(pProHW,pProslic->channel,74);
220  WriteReg(pProHW,pProslic->channel,74,daaEnableSave|0x01);
221  data = ReadReg(pProHW,pProslic->channel,SI3217X_COM_REG_PCMTXHI);
222  DEBUG_PRINT(pProslic, "%sisVerifiedProslic : REG VAL = %02x\n", LOGPRINT_PREFIX,
223  data);
224  if ( data == 0x40 ) /* Always 0x40 if lineside device is not enabled yet*/
225  {
226  pProslic->channelType = DAA;
227  WriteReg(pProHW,pProslic->channel,74,daaEnableSave);
228  return RC_NONE; /* Even though this not a SLIC, mark it as a valid part */
229  }
230  } /* DAA not enabled */
231  /* Not a VDAA - could be ProSLIC or empty channel */
232  data = ReadReg(pProHW,pProslic->channel,
233  SI3217X_COM_REG_PCMMODE); /* always 0x05 after reset */
234  if ( data == 0x05)
235  {
236  pProslic->deviceId->chipRev = id & 0x7;
237  pProslic->deviceId->chipType = getChipType(id);
238 
239  if(pProslic->deviceId->chipType == UNSUPPORTED_PART_NUM)
240  {
241 #ifdef ENABLE_DEBUG
242  LOGPRINT("%sregister 0 read = 0x%02X\n", LOGPRINT_PREFIX, id);
243 #endif
244  return RC_SPI_FAIL;
245  }
246  else
247  {
248  return RC_NONE;
249  }
250 
251  pProslic->channelType = PROSLIC;
252  returnCode = RC_NONE;
253  }
254  else
255  {
256  pProslic->channelType = UNKNOWN;
257  returnCode = RC_CHANNEL_TYPE_ERR;
258 #ifdef ENABLE_DEBUG
259  LOGPRINT("%sregister 11 read = 0x%02X\n", LOGPRINT_PREFIX, data);
260 #endif
261  }
262  return returnCode;
263 }

266 /*
267 ** Function: Si3217x_PowerUpConverter
268 **
269 ** Description:
270 ** Powers all DC/DC converters sequentially with delay to minimize
271 ** peak power draw on VDC.
272 **
273 ** Returns:
274 ** int (error)
275 **
276 */
277 int Si3217x_PowerUpConverter(proslicChanType_ptr pProslic)
278 {
279  errorCodeType error = RC_DCDC_SETUP_ERR;
280  int32 vbath,vbat;
281  uInt8 reg = 0;
282  int timer = 0;
283 
284 
285  if(pProslic->channelType != PROSLIC)
286  {
287  return RC_CHANNEL_TYPE_ERR;
288  }
289 
290  /*
291  ** - powerup digital dc/dc w/ OV clamping and shutdown
292  ** - delay
293  ** - verify no short circuits by looking for vbath/2
294  ** - clear dcdc status
295  ** - switch to analog converter with OV clamping only (no shutdown)
296  ** - select analog dcdc and disable pwrsave
297  ** - delay
298  */
299 
300  WriteReg(pProHW,pProslic->channel,PROSLIC_REG_LINEFEED,
301  LF_FWD_OHT); /* Force out of pwrsave mode if called in error */
302  WriteReg(pProHW,pProslic->channel,PROSLIC_REG_LINEFEED,
303  LF_OPEN); /* Ensure open line before powering up converter */
304  reg = ReadReg(pProHW,pProslic->channel,SI3217X_COM_REG_ENHANCE);
305  WriteReg(pProHW,pProslic->channel,SI3217X_COM_REG_ENHANCE,
306  reg&0x07); /* Disable powersave mode */
307 
308  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_PD_DCDC,
309  0x700000L); /* In case OV or UV previously occurred */
310 
311  /*
312  ** Setup converter drive polarity and charge pump enable
313  ** based on revision, bom, and converter type
314  */
315  if(pProslic->deviceId->chipRev == SI3217X_REVB)
316  {
317 #ifndef SI3217X_REVC_ONLY
318  error = Si3217x_RevB_ConverterSetup(pProslic);
319 #else
320  error = RC_UNSUPPORTED_DEVICE_REV;
321 #endif
322  }
323  else /* Add revC option here */
324  {
325 #ifndef SI3217X_REVB_ONLY
326  error = Si3217x_RevC_ConverterSetup(pProslic);
327 #else
328  error = RC_UNSUPPORTED_DEVICE_REV;
329 #endif
330  }
331 
332  if(error != RC_NONE)
333  {
334  DEBUG_PRINT(pProslic, "%sChannel %d : DCDC initialization failed\n",
335  LOGPRINT_PREFIX, pProslic->channel);
336  return error;
337  }
338 
339  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_PD_DCDC,0x600000L);
340  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_PD_BIAS,0x200000L);
341  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_PD_HVIC,0x200000L);
342  Delay(pProTimer,50);
343  vbath = ReadRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_VBATH_EXPECT);
344  vbat = ReadRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_MADC_VBAT);
345  if(vbat & 0x10000000L)
346  {
347  vbat |= 0xF0000000L;
348  }
349  if(vbat < (vbath / 2))
350  {
351  pProslic->channelEnable = 0;
352  error = RC_VBAT_UP_TIMEOUT;
353  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_PD_DCDC,
354  0x300000L); /* shutdown converter */
355  DEBUG_PRINT(pProslic,
356  "%sChannel %d : DCDC Short Circuit Failure - disabling channel VBAT = %d.%d\n",
357  LOGPRINT_PREFIX, pProslic->channel, (int)((vbat/SCALE_V_MADC)/1000),
358  SIVOICE_ABS(((vbat/SCALE_V_MADC) - (vbat/SCALE_V_MADC)/1000*1000)));
359  return error;
360  }
361  else /* Enable analog converter */
362  {
363  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_DCDC_STATUS,0L);
364  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_PD_DCDC,0x400000L);
365  WriteReg(pProHW,pProslic->channel,SI3217X_COM_REG_ENHANCE,
366  reg); /* Restore ENHANCE */
367  Delay(pProTimer,50);
368  }
369 
370  /*
371  ** - monitor vbat vs expected level (VBATH_EXPECT)
372  */
373  vbath = ReadRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_VBATH_EXPECT);
374  do
375  {
376  vbat = ReadRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_MADC_VBAT);
377  if(vbat & 0x10000000L)
378  {
379  vbat |= 0xF0000000L;
380  }
381  Delay(pProTimer,10);
382  }
383  while((vbat < (vbath - COMP_5V))
384  &&(timer++ < SI3217X_TIMEOUT_DCDC_UP)); /* 2 sec timeout */
385 
386  DEBUG_PRINT (pProslic, "%sChannel %d : VBAT Up = %d.%d v\n",
387  LOGPRINT_PREFIX,
388  pProslic->channel,(int)((vbat/SCALE_V_MADC)/1000),
389  SIVOICE_ABS(((vbat/SCALE_V_MADC) - (vbat/SCALE_V_MADC)/1000*1000)));
390  if(timer > SI3217X_TIMEOUT_DCDC_UP)
391  {
392  /* Error handling - shutdown converter, disable channel, set error tag */
393  pProslic->channelEnable = 0;
394  error = RC_VBAT_UP_TIMEOUT;
395  WriteRAM(pProHW,pProslic->channel,SI3217X_COM_RAM_PD_DCDC,
396  0x900000L); /* shutdown converter */
397  DEBUG_PRINT (pProslic, "%sChannel %d : DCDC Power up timeout : Status=0x%08X\n",
398  LOGPRINT_PREFIX, pProslic->channel, ReadRAM(pProHW,pProslic->channel,
399  (int)SI3217X_COM_RAM_DCDC_STATUS));
400  }
401 
402  return error;
403 }

451 ** Function: Si3217x_Init_with_Options
452 **
453 ** Description:
454 ** - probe SPI to establish daisy chain length
455 ** - load patch
456 ** - initialize general parameters
457 ** - calibrate madc
458 ** - bring up DC/DC converters
459 ** - calibrate everything except madc & lb
460 **
461 ** Input Parameters:
462 ** pProslic: pointer to PROSLIC object array
463 ** fault: error code
464 **
465 ** Return:
466 ** error code
467 */
468 
469 int Si3217x_Init_with_Options (proslicChanType_ptr *pProslic, int size,
470  initOptionsType init_opt)
471 {
472  /*
473  ** This function will initialize the chipRev and chipType members in pProslic
474  ** as well as load the initialization structures.
475  */
476  uInt8 data;
477  uInt8 calSetup[] = {0x00, 0x00, 0x01, 0x80};
478  int j,k;
479  int device_count = 0;
480  const proslicPatch *patch;
481  uInt8 status;
482 
483  LOGPRINT("%s(%d) size = %d init_opt = %d\n", __FUNCTION__, __LINE__, size,
484  init_opt);
485 
486  /* reset error code */
487  for(k = 0; k < size; k++)
488  {
489  pProslic[k]->error = RC_NONE;
490  }
491 
492  if( (init_opt == INIT_REINIT) || (init_opt == INIT_SOFTRESET) )
493  {
494  ProSLIC_ReInit_helper(pProslic, size, init_opt, SI3217X_CHAN_PER_DEVICE);
495  /* for single channel devices, we need do a full restore.. */
496  if(init_opt == INIT_REINIT)
497  {
498  init_opt = 0;
499  }
500  }
501 
502 
503  if( (init_opt != INIT_REINIT) )
504  {
505  if( (SiVoice_IdentifyChannels(pProslic, size, &device_count, NULL) != RC_NONE)
506  ||(device_count == 0) )
507  {
508  DEBUG_PRINT(*pProslic, "%s: failed to detect any ProSLICs\n", LOGPRINT_PREFIX);
509  return RC_SPI_FAIL;
510  }
511 
512  /*
513  ** Probe each channel and enable all channels that respond
514  */
515  for (k=0; k<size; k++)
516  {
517  if ((pProslic[k]->channelEnable)
518  &&(pProslic[k]->channelType == PROSLIC))
519  {
520  if ( (ProSLIC_VerifyMasterStat(pProslic[k]) != RC_NONE)
521  || (ProSLIC_VerifyControlInterface(pProslic[k]) != RC_NONE) )
522  {
523  pProslic[k]->channelEnable = 0;
524  pProslic[k]->error = RC_SPI_FAIL;
525  DEBUG_PRINT(*pProslic, "%s: SPI communications or PCLK/FS failure\n", LOGPRINT_PREFIX);
526  return pProslic[k]->error; /* Halt init if SPI fail */
527  }
528  }
529  }
530  } /* init_opt !REINIT*/
531 
532 
533  if( (init_opt != INIT_NO_PATCH_LOAD ) && (init_opt != INIT_REINIT) )
534  {
535  /*
536  ** Load patch (load on every channel since single channel device)
537  */
538  for (k=0; k<size; k++)
539  {
540  if ((pProslic[k]->channelEnable)&&(pProslic[k]->channelType == PROSLIC))
541  {
542 
543  /* Select Patch*/
544  if (pProslic[k]->deviceId->chipRev == SI3217X_REVB )
545  {
546 #ifndef SI3217X_REVC_ONLY
547  status = (uInt8) Si3217x_RevB_SelectPatch(pProslic[k],&patch);
548 #else
549  status = RC_UNSUPPORTED_DEVICE_REV;
550 #endif
551  }
552  else if (pProslic[k]->deviceId->chipRev == SI3217X_REVC )
553  {
554 #ifndef SI3217X_REVB_ONLY
555  status = (uInt8) Si3217x_RevC_SelectPatch(pProslic[k],&patch);
556 #else
557  status = RC_UNSUPPORTED_DEVICE_REV;
558 #endif
559  }
560  else
561  {
562  DEBUG_PRINT (pProslic[k], "%sChannel %d : Unsupported Device Revision (%d)\n",
563  LOGPRINT_PREFIX,
564  pProslic[k]->channel,pProslic[k]->deviceId->chipRev );
565  pProslic[k]->channelEnable = 0;
566  pProslic[k]->error = RC_UNSUPPORTED_DEVICE_REV;
567  return RC_UNSUPPORTED_DEVICE_REV;
568  }
569 
570  data = 1; /* Use this as a flag to see if we need to load the patch */
571  /* Check if the patch is different than we expect */
572  if(init_opt == INIT_SOFTRESET)
573  {
574  uInt32 patchData;
575  patchData = pProslic[k]->ReadRAMX(pProslic[k]->pProHWX, pProslic[k]->channel,
576  PROSLIC_RAM_PATCHID);
577 
578  if( patchData == patch->patchSerial)
579  {
580  data = 0;
581  }
582  else
583  {
584  init_opt = INIT_NO_OPT;
585  }
586  }
587 
588  /* Load Patch */
589  if(status == RC_NONE)
590  {
591  if(data == 1)
592  {
593  Si3217x_LoadPatch(pProslic[k],patch);
594 #ifndef DISABLE_VERIFY_PATCH
595  data = (uInt8)Si3217x_VerifyPatch(pProslic[k],patch);
596  if (data != RC_NONE)
597  {
598  DEBUG_PRINT(pProslic[k], "%sChannel %d : Patch verification failed (%d)\n",
599  LOGPRINT_PREFIX, k, data);
600  pProslic[k]->channelEnable=0;
601  pProslic[k]->error = RC_PATCH_ERR;
602  return data;
603  }
604 #endif
605  }
606  }
607  else
608  {
609  return status;
610  }
611 
612  /* Optional Patch Verification */
613  } /* channel == PROSLIC */
614  } /* for all channles */
615  }/* init_opt - need to reload patch */
616 
617  /*
618  ** Load general parameters - includes all BOM dependencies
619  */
620  if(init_opt != INIT_SOFTRESET)
621  {
622  for (k=0; k<size; k++)
623  {
624  if ((pProslic[k]->channelEnable)&&(pProslic[k]->channelType == PROSLIC))
625  {
626  if(pProslic[k]->deviceId->chipRev == SI3217X_REVB)
627  {
628 #ifndef SI3217X_REVC_ONLY
629  Si3217x_RevB_GenParamUpdate(pProslic[k],INIT_SEQ_PRE_CAL);
630 #endif
631  }
632  else
633  {
634 #ifndef SI3217X_REVB_ONLY
635  Si3217x_RevC_GenParamUpdate(pProslic[k],INIT_SEQ_PRE_CAL);
636 #endif
637  }
638  pProslic[k]->WriteRAMX(pProslic[k]->pProHWX,pProslic[k]->channel,
639  SI3217X_COM_RAM_IRING_LIM,SI3217X_IRING_LIM_MAX);
640  }
641  }
642  }
643 
644  if((init_opt != INIT_NO_CAL)
645  && (init_opt != INIT_SOFTRESET)) /* Must recal on single channel devices */
646  {
647  /*
648  ** Calibrate (madc offset)
649  */
650  ProSLIC_Calibrate(pProslic,size,calSetup,TIMEOUT_MADC_CAL);
651  }/* init_opt */
652 
653  /*
654  ** Bring up DC/DC converters sequentially to minimize
655  ** peak power demand on VDC
656  */
657  for (k=0; k<size; k++)
658  {
659  if ((pProslic[k]->channelEnable)&&(pProslic[k]->channelType == PROSLIC))
660  {
661  pProslic[k]->error = Si3217x_PowerUpConverter(pProslic[k]);
662  }
663  }
664 
665  if((init_opt != INIT_NO_CAL) && (init_opt != INIT_SOFTRESET))
666  {
667  /*
668  ** Calibrate remaining cals (except madc, lb)
669  ** ASSUME: same chip revs across selection... for the given
670  ** general configuration, this is usually a safe assumption.
671  */
672  data = (*pProslic)->deviceId->chipRev;
673  j = 0;
674  /* We need to iterate through and find any mismatches in revs, stop
675  when we do and do a calibrate with the prior block's value.
676  */
677  for(k= 0; k < size; k++)
678  {
679  if(pProslic[k]->deviceId->chipRev != data)
680  {
681  if(data == SI3217X_REVB)
682  {
683  calSetup[1] = SI3217XB_CAL_STD_CALR1;
684  calSetup[2] = SI3217XB_CAL_STD_CALR2;
685  }
686  else /* ASSUME: Rev C or greater */
687  {
688  calSetup[1] = SI3217XC_CAL_STD_CALR1;
689  calSetup[2] = SI3217XC_CAL_STD_CALR2;
690  }
691  ProSLIC_Calibrate(&pProslic[j],(k-j),calSetup,TIMEOUT_GEN_CAL);
692  j = k;
693  data = pProslic[k]->deviceId->chipRev;
694  }
695  }
696 
697  /* Handle the last block of chips needing to be calibrated */
698  if(data == SI3217X_REVB)
699  {
700  calSetup[1] = SI3217XB_CAL_STD_CALR1;
701  calSetup[2] = SI3217XB_CAL_STD_CALR2;
702  }
703  else /* ASSUME: Rev C or greater */
704  {
705  calSetup[1] = SI3217XC_CAL_STD_CALR1;
706  calSetup[2] = SI3217XC_CAL_STD_CALR2;
707  }
708  ProSLIC_Calibrate(&pProslic[j],(size-j),calSetup,TIMEOUT_GEN_CAL);
709  }
710 
711  /*
712  ** Apply post calibration general parameters
713  */
714  if(init_opt != INIT_SOFTRESET)
715  {
716  for (k=0; k<size; k++)
717  {
718  if ((pProslic[k]->channelEnable)&&(pProslic[k]->channelType == PROSLIC))
719  {
720  if(pProslic[k]->deviceId->chipRev == SI3217X_REVB)
721  {
722 #ifndef SI3217X_REVC_ONLY
723  Si3217x_RevB_GenParamUpdate(pProslic[k],INIT_SEQ_POST_CAL);
724 #endif
725  }
726  else
727  {
728 #ifndef SI3217X_REVB_ONLY
729  Si3217x_RevC_GenParamUpdate(pProslic[k],INIT_SEQ_POST_CAL);
730 #endif
731  }
732  }
733  }
734  } /* ! SOFTRESET */
735 
736  /* Restore linefeed state after initialization for REINIT/SOFTRESET */
737  if( (init_opt == INIT_REINIT) || (init_opt == INIT_SOFTRESET) )
738  {
739  for(k = 0; k < size; k++)
740  {
741  pProslic[k]->WriteRegX(pProslic[k]->pProHWX,pProslic[k]->channel,
742  PROSLIC_REG_LINEFEED,pProslic[k]->scratch);
743  }
744  }

747  ** If any channel incurred a non-fatal error, return
748  ** RC_NON_FATAL_INIT_ERR to trigger user to read each channel's
749  ** error status
750  */
751  for (k=0; k<size; k++)
752  {
753  if(pProslic[k]->error != RC_NONE)
754  {
755  return RC_NON_FATAL_INIT_ERR;
756  }
757  }
